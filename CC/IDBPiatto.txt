package unisa.gps.etour.repository;

import java.sql.SQLException;
import java.util.ArrayList;

import unisa.gps.etour.bean.DishBean;

/**
 * Interface for the management of food in the database
 *
 * @ Author Joseph Martone
 * @ Version 0.1 2007 eTour Project - Copyright by SE @ SA Lab DMI University
 * Of Salerno
 */
public interface FoodDBInterface {

    /**
     * Add a dish
     *
     * @ Param pDish dish to add
     */
    public boolean insertDish(DishBean pDish) throws SQLException;

    /**
     * Modify the data in a dish in the database
     *
     * @ Param pDish data of the dish to be inserted in the database
     * @ Return True if there 'was a modified false otherwise
     */
    public boolean editDish(DishBean pDish) throws SQLException;

    /**
     * Delete a dish from the database
     *
     * @ Param pDishId ID of the dish to delete
     * @Throws SQLException
     * @ Return True if and 'was deleted false otherwise
     */
    public boolean deleteDish(int pDishId) throws SQLException;

    /**
     * Returns a list of dishes on a menu
     *
     * @ Param menuId ID of the menu
     * @Throws SQLException
     * @ Return list of dishes in the menu
     */
    public ArrayList<DishBean> getDishes(int menuId) throws SQLException;
}
